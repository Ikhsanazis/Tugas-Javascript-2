// Buat 10 method function pada javascript

let contoh = "ikhsan azis pane";
//1 mengubah karakter menjadi huruf kecil
console.log(contoh.toLowerCase()); //output ikhsan azis pane
// 2 mengubah karakter menjadi huruf besar
console.log(contoh.toUpperCase()); //output IKHSAN AZIS PANE

const input = "    M. Ikhsan Azis Pane   ";
// 3 merapikan karakter spasi
console.log(input.trim()); //output M. Ikhsan Azis Pane

const value = "1.02345";
//4 mengubah nilai menjadi integer
const newValue = parseInt(value);
console.log(newValue); //output 1
//5 mengubah nilai menjadi number
const newValue2 = Number(value);
console.log(newValue2); //ouput 1.02345
//6 menampilkan tipe data
console.log(typeof newValue); //output number

const array = [];
//7 menambahkan komponen array setelah index terakhir
array.push("azis");
array.push("ikhsan");

//8 menghapus komponen array indeks terakhir
array.pop();
console.log(array); //ouput [ 'azis' ]
//9 menambahkan komponen array indeks pertama
array.unshift("Muhammad");

//10 menghapus komponen array pada indeks pertama
array.shift();
